#include <Wire.h>
#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);

// Pin
const int voltageSensorPin = A0;  // Voltage measurement

void setup() {
  Serial.begin(9600);

  lcd.init();
  lcd.backlight();

  lcd.setCursor(1, 0);
  lcd.print("Footstep Power");
  lcd.setCursor(3, 1);
  lcd.print("Generator");
  delay(2000);

  lcd.clear();
  lcd.setCursor(1, 0);
  lcd.print("VOLTAGE OUTPUT");
  lcd.setCursor(0, 1);
  lcd.print("voltage: ---mV");
}

void loop() {

  // -------- VOLTAGE READING --------
  int raw = analogRead(voltageSensorPin);
  int milliVoltage = (raw * 5000UL) / 1023;

  // -------- LCD UPDATE --------
  lcd.setCursor(0, 1);
  lcd.print("Voltage: ");
  lcd.print(milliVoltage);
  lcd.print(" mV   ");   // Clear old digits

  // -------- SERIAL OUTPUT --------
  Serial.print("Voltage: ");
  Serial.print(milliVoltage);
  Serial.println(" mV");

  delay(150);
}
